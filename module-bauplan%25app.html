Module bauplan.app

Application instance

var App = require("bauplan.app");

or as part of the Bauplan bundle

var Bauplan = require("bauplan");

To set or get the app locale

App.locale("fr");
var loc = App.locale(); // "fr"

To (re-)render the entire app

App.render();
